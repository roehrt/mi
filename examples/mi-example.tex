\newcommand{\setupnodes}{
    \begin{scope}[every node/.style={vertex}]
        \node (a) at (0,8) {a};
        \node (b) at (9,9) {b};
        \node (c) at (8,6) {c};
        \node (d) at (6,6) {d};
        \node (e) at (4,5) {e};
        \node (f) at (0,4) {f};
        \node (g) at (9,3) {g};
    \end{scope}
}
\newcommand{\setupedges}{
    \begin{scope}[-Stealth]
        \draw (a) to (b);
        \draw (a) to (d);
        \draw (a) to (e);
        \draw (a) to (f);
        \draw (b) to (c);
        \draw (b) to (d);
        \draw (b) to (g);
        \draw (c) to (d);
        \draw (d) to (e);
        \draw (d) to (g);
        \draw (e) to (f);
    \end{scope}
}
